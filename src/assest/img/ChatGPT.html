import React from 'react';
import { render, screen, fireEvent } from '@testing-library/react';
import '@testing-library/jest-dom/extend-expect';
import NavBar from './NavBar';
import { useSession } from 'next-auth/react';
import { useRouter } from 'next/navigation';
import useResponsive from '@/hooks/useResponsive';

// Mock necessary modules
jest.mock('next-auth/react');
jest.mock('next/navigation', () => ({
  useRouter: jest.fn(),
}));
jest.mock('@/hooks/useResponsive');

describe('NavBar Component', () => {
  let mockPush: jest.Mock;

  beforeEach(() => {
    mockPush = jest.fn();
    (useRouter as jest.Mock).mockReturnValue({
      push: mockPush,
      back: jest.fn(),
      forward: jest.fn(),
      pathname: '/',
    });
  });

  afterEach(() => {
    jest.clearAllMocks();
  });

  it('renders mobile version correctly', () => {
    (useResponsive as jest.Mock).mockReturnValue({
      isMobile: true,
      isTablet: false,
      isDesktop: false,
    });

    render(<NavBar />);

    expect(screen.getByTestId('navbarMobile')).toBeInTheDocument();
    expect(screen.getByTestId('homeBtn')).toBeInTheDocument();
  });

  it('renders desktop version correctly', () => {
    (useResponsive as jest.Mock).mockReturnValue({
      isMobile: false,
      isTablet: false,
      isDesktop: true,
    });

    render(<NavBar />);

    expect(screen.getByTestId('nav-desktop')).toBeInTheDocument();
  });

  it('handles mobile navigation button clicks', () => {
    (useResponsive as jest.Mock).mockReturnValue({
      isMobile: true,
      isTablet: false,
      isDesktop: false,
    });

    render(<NavBar />);

    const homeButton = screen.getByTestId('homeBtn');
    fireEvent.click(homeButton);

    expect(mockPush).toHaveBeenCalledWith('/');
  });

  it('handles desktop back and forward button clicks', () => {
    (useResponsive as jest.Mock).mockReturnValue({
      isMobile: false,
      isTablet: false,
      isDesktop: true,
    });

    render(<NavBar />);

    const backButton = screen.getByRole('button', { name: /back/i });
    const forwardButton = screen.getByRole('button', { name: /forward/i });

    fireEvent.click(backButton);
    fireEvent.click(forwardButton);

    expect(mockPush).not.toHaveBeenCalled(); // Ensure push isn't called with these buttons
  });

  it('displays the user menu when logged in', () => {
    (useResponsive as jest.Mock).mockReturnValue({
      isMobile: false,
      isTablet: false,
      isDesktop: true,
    });

    (useSession as jest.Mock).mockReturnValue({
      data: {
        user: {
          name: 'John Doe',
          id: '123',
        },
      },
      status: 'authenticated',
    });

    render(<NavBar />);

    const avatarButton = screen.getByRole('button', { name: /john doe/i });
    fireEvent.click(avatarButton);

    expect(screen.getByText(/Tài khoản/i)).toBeInTheDocument();
    expect(screen.getByText(/Đăng xuất/i)).toBeInTheDocument();
  });

  it('does not display the user menu when logged out', () => {
    (useResponsive as jest.Mock).mockReturnValue({
      isMobile: false,
      isTablet: false,
      isDesktop: true,
    });

    (useSession as jest.Mock).mockReturnValue({
      data: null,
      status: 'unauthenticated',
    });

    render(<NavBar />);

    expect(screen.queryByRole('button', { name: /john doe/i })).not.toBeInTheDocument();
  });

  it('handles logout action', () => {
    (useResponsive as jest.Mock).mockReturnValue({
      isMobile: false,
      isTablet: false,
      isDesktop: true,
    });

    (useSession as jest.Mock).mockReturnValue({
      data: {
        user: {
          name: 'John Doe',
          id: '123',
        },
      },
      status: 'authenticated',
    });

    render(<NavBar />);

    const avatarButton = screen.getByRole('button', { name: /john doe/i });
    fireEvent.click(avatarButton);

    const logoutButton = screen.getByText(/Đăng xuất/i);
    fireEvent.click(logoutButton);

    expect(mockPush).toHaveBeenCalledWith('/login');
  });
});

  3. Session Handling:
